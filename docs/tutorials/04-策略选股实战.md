### 策略的本质

```

│   │                                                         │   │
```

### 实战示例: 金叉策略

```python
# 文件: instock/core/strategy/golden_cross.py

"""
MACD金叉策略

当MACD指标出现金叉时选中股票
"""

import pandas as pd
import talib

__strategy_name__ = "MACD金叉"


def check(stock_data: pd.DataFrame, end_date: str = None) -> bool:
    """
    检查是否出现MACD金叉
    
    金叉定义: DIF从下方穿越DEA
    """
    
    # 数据检查
    if len(stock_data) < 35:  # MACD需要至少35天数据
        return False
    
    # 获取收盘价序列
    close = stock_data['close'].values
    
    # 计算MACD
    dif, dea, macd = talib.MACD(
        close,
        fastperiod=12,
        slowperiod=26,
        signalperiod=9
    )
    
    # 判断金叉
    # 今天: DIF > DEA
    # 昨天: DIF < DEA
    today_cross = dif[-1] > dea[-1]
    yesterday_below = dif[-2] < dea[-2]
    
    # 金叉 = 今天在上方 且 昨天在下方
    is_golden_cross = today_cross and yesterday_below
    
    return is_golden_cross
```

```python
# instock/core/strategy/two_days_up.py

"""连涨两天策略"""

__strategy_name__ = "连涨两天"

def check(stock_data, end_date=None):
    if len(stock_data) < 3:
        return False
    
    # 今天上涨
    today_up = stock_data.iloc[-1]['close'] > stock_data.iloc[-1]['open']
    # 昨天上涨
    yesterday_up = stock_data.iloc[-2]['close'] > stock_data.iloc[-2]['open']
    
    return today_up and yesterday_up
```
